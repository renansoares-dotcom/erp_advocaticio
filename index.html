-- ============================================================================ -- Migração 024: Diário Oficial (publicações) -- ============================================================================ -- Camada manual do módulo de Diário Oficial: cada publicação é cadastrada -- por uma pessoa do escritório (copiando/colando o texto do diário), com -- vínculo opcional a um processo (pode ficar "solta" até alguém localizar -- o processo certo), responsável pela providência, e um fluxo de status. -- -- Pensado para, no futuro, ser alimentado automaticamente por um conector -- de monitoramento (Codilo/Escavador/JusBrasil/etc) — quando isso existir, -- o conector apenas insere linhas nesta mesma tabela (com origem='automatica'), -- sem mudar nada do que já está em uso. -- ============================================================================ create table if not exists public.publicacoes_diario ( id uuid primary key default gen_random_uuid(), escritorio_id uuid not null references public.escritorios(id) on delete cascade, processo_id uuid references public.processos(id) on delete set null, -- opcional: pode ficar "solta" cliente_id uuid references public.clientes(id) on delete set null, -- opcional: ajuda a localizar mesmo sem processo diario text, -- nome do diário/veículo (ex: "DJE-SP", "DOU") data_publicacao date not null, numero_processo_publicado text, -- número do processo como veio no diário (pode não bater 100% com o cadastrado) conteudo text not null, -- texto da publicação (colado manualmente, por enquanto) responsavel text, -- nome livre, mesmo padrão usado em tarefas/cartões status text not null default 'nova' check (status in ('nova','lida','providenciada','concluida')), origem text not null default 'manual' check (origem in ('manual','automatica')), criado_por uuid references auth.users(id), criado_em timestamptz not null default now(), atualizado_em timestamptz not null default now() ); comment on table public.publicacoes_diario is 'Publicações do Diário Oficial — cadastro manual por enquanto, com vínculo opcional a processo/cliente. Campo "origem" prepara para automação futura via API de monitoramento.'; create index if not exists idx_publicacoes_diario_escritorio on public.publicacoes_diario (escritorio_id, data_publicacao desc); create index if not exists idx_publicacoes_diario_processo on public.publicacoes_diario (processo_id); create index if not exists idx_publicacoes_diario_status on public.publicacoes_diario (escritorio_id, status) where status in ('nova','lida'); alter table public.publicacoes_diario enable row level security; drop policy if exists publicacoes_diario_select on public.publicacoes_diario; create policy publicacoes_diario_select on public.publicacoes_diario for select to authenticated using (escritorio_id in (select public.escritorio_ids())); drop policy if exists publicacoes_diario_insert on public.publicacoes_diario; create policy publicacoes_diario_insert on public.publicacoes_diario for insert to authenticated with check ( escritorio_id in (select public.escritorio_ids()) and (processo_id is null or exists (select 1 from processos p where p.id = processo_id and p.escritorio_id = publicacoes_diario.escritorio_id)) and (cliente_id is null or exists (select 1 from clientes c where c.id = cliente_id and c.escritorio_id = publicacoes_diario.escritorio_id)) ); drop policy if exists publicacoes_diario_update on public.publicacoes_diario; create policy publicacoes_diario_update on public.publicacoes_diario for update to authenticated using (escritorio_id in (select public.escritorio_ids())) with check ( escritorio_id in (select public.escritorio_ids()) and (processo_id is null or exists (select 1 from processos p where p.id = processo_id and p.escritorio_id = publicacoes_diario.escritorio_id)) and (cliente_id is null or exists (select 1 from clientes c where c.id = cliente_id and c.escritorio_id = publicacoes_diario.escritorio_id)) ); drop policy if exists publicacoes_diario_delete on public.publicacoes_diario; create policy publicacoes_diario_delete on public.publicacoes_diario for delete to authenticated using (escritorio_id in (select public.escritorio_ids())); drop trigger if exists trg_publicacoes_diario_atualizado_em on public.publicacoes_diario; create trigger trg_publicacoes_diario_atualizado_em before update on public.publicacoes_diario for each row execute function public.set_atualizado_em();
